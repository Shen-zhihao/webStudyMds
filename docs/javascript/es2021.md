##### Promise.any
```javascript
与 Promise.all 类似，Promise.any 也接受一个 Promise 的数组。当其中任何一个 Promise 完成（fullfill）时，就返回那个已经有完成值的 Promise。
如果所有的 Promise 都拒绝（reject），则返回一个拒绝的 Promise，该 Promise 的返回值是一个 AggregateError 对象。我们可以把 Promise.any 理解成 Promise.all 的相反操作。
  Promise.any(promises).then(
    (first) => {
      // 任何一个 Promise 完成了
    },
    (error) => {
      // 所有的 Promise 都拒绝了
    }
  );
```

##### 逻辑符号简写
```javascript
   a ||= b;// 等同于 a = a || b  如果 a 为真则返回 a，如果 a 为假则返回 b
   c &&= d;// 等同于 c = c && d  如果 a 为真，则返回 b , 如果 a 为假，则返回 a
   e ??= f;// 等同于 e = e ?? f  如果 e 为 null或未定义，则返回 f；如果e为真，则返回e。
```

##### Object.hasOwn()方法，检查某个对象自身是否拥有某个属性
```javascript
	const obj = Object.create(null);
	obj.color = 'green';
	obj.age = 2;
	obj.hasOwnProperty = () => false;

	console.log(Object.hasOwn(obj, 'color')); // true
	console.log(Object.hasOwn(obj, 'name'));  // false  基本等价于obj.hasOwnProperty('name')
```

##### 数组支持逆序查找findLast()和findLastIndex(）
```javascript

	const lastEven = nums.findLast((num) => num % 2 === 0);
	const lastEvenIndex = nums.findLastIndex((num) => num % 2 === 0);

	console.log(lastEven); // 10
	console.log(lastEvenIndex); // 4
```

##### Symbol 作为 WeakMap 的键
```javascript
在这之前，WeakMap仅允许对象作为键值，新特性更容易创建和共享key。

var map = new WeakMap(); // 创建一个弱映射
function useSymbol(symbol){
    doSomethingWith(symbol);
    var called = map.get(symbol) || 0

上面的例子允许从外部调用者调用计数器，并在不再有引用时释放映射条目。
代码本身无法知道何时不再需要引用，如果使用普通的 Map，将会导致内存泄漏。
这是因为即使在调用它的客户端不再需要引用时，代码仍然会保留对该引用的持有。
在这种情况下使用 WeakMap，可以确保垃圾回收在不再存在对键符号的引用时删除映射条目。
```

## ES2024
##### Promise.withResolvers()
```javascript
// Promise.withResolvers() 允许创建一个新的 Promise，并同时获得 resolve 和 reject 函数。

// old
let resolve, reject;  
const promise = new Promise((res, rej) => {  
    resolve = res;  
    reject = rej;  
});

// new 
const { promise, resolve, reject } = Promise.withResolvers();  
  
// 在这里可以使用 resolve 和 reject 函数  
setTimeout(() => resolve('成功！'), 8000);  
  
promise.then(value => {  
    console.log(value); // 输出: 成功！  
});
```

## ES2025
##### Promise.try()
```typescript
一、核心作用
  Promise.try() 是 ES2025 新增的静态方法，主要解决 ​异步入口统一化​ 和 ​错误捕获一致性​ 两大问题：
​统一处理同步/异步函数​
  将任何函数（同步/异步、可能抛错/返回 Promise）自动包装为 Promise，消除类型判断需求
​自动捕获同步错误​
  传统 new Promise 需要手动 try-catch，而 Promise.try() 自动捕获同步异常并转为 rejected Promise
​消除微任务延迟​
  直接执行同步函数而非放入微任务队列，避免不必要的异步化

// 基本语法
Promise.try(executorFn, [arg1, arg2, ...]);

// 支持环境：现代浏览器（Chrome 110+/Firefox 105+/Safari 16+）

// 参数说明
- executorFn: 需要包装的函数（可同步/异步）
- arg1...: 传递给 executorFn 的参数
```

```typescript
const syncTask = () => {
  console.log("立即执行同步任务");
  return "同步结果";
};

Promise.try(syncTask)
  .then(result => console.log(result)) // 输出：同步结果
  .catch(() => {});
```

```typescript
const asyncTask = () => new Promise(resolve => {
  setTimeout(() => resolve("异步结果"), 1000);
});

Promise.try(asyncTask)
  .then(result => console.log(result)) // 1秒后输出：异步结果
  .catch(() => {});
```

```typescript
const riskyTask = () => {
  if (Math.random() > 0.5) throw new Error("随机错误"); // 同步错误
  return "安全值";
};

Promise.try(riskyTask)
  .then(result => console.log(result))
  .catch(err => console.error("捕获到错误:", err.message));
```

```typescript
function fetchData(url) {
  return Promise.try(() => {
    // 同步校验 URL
    if (!url.startsWith("http")) throw new Error("无效URL");
    
    // 异步请求
    return fetch(url).then(res => {
      if (!res.success) throw new Error("请求失败");
      return res.json();
    });
  });
}

fetchData("/api/data")
  .then(data => console.log(data))
  .catch(err => console.error(err));
```

```typescript
Promise.try = function(executor, ...args) {
  return new Promise((resolve, reject) => {
    try {
      // 立即执行 executor 函数
      const result = executor(...args);
      // 处理返回值
      Promise.resolve(result).then(resolve, reject);
    } catch (err) {
      // 捕获同步错误
      reject(err);
    }
  });
};
```
